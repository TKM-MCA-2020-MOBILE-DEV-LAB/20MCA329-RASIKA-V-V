package com.example.customer;

import android.content.ContentValues;
import android.content.Context;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;

import androidx.annotation.Nullable;

public class Database extends SQLiteOpenHelper {

   public static final String CUSTDATA = "Custdata";
   public static final String CUST_ID = "id";
   public static final String CUSTOMER_NAME = "CustomerName";
   public static final String CUST_AGE = "Age";
   public static final String ACTIVE_USER = "Activeuser";

   public Database(@Nullable Context context) {
       super(context, "Cust.db", null, 1);
   }

   @Override
   public void onCreate(SQLiteDatabase sqLiteDatabase) {
       String s = "create table " + CUSTDATA + "(" + CUST_ID + " integer primary key autoincrement," + CUSTOMER_NAME + " text," + CUST_AGE + " int," + ACTIVE_USER + " bool)";
       sqLiteDatabase.execSQL(s);

   }

   @Override
   public void onUpgrade(SQLiteDatabase sqLiteDatabase, int i, int i1) {

   }

   public Boolean add(Customer cus) {
       SQLiteDatabase db = this.getWritableDatabase();
       ContentValues cv = new ContentValues();
       cv.put(CUSTOMER_NAME, cus.getCname());
       cv.put(CUST_AGE, cus.getAge());
       cv.put(ACTIVE_USER, cus.isActuser());
       long insert=db.insert(CUSTDATA,null,cv);
       if(insert==-1){
           return false;
       }
       else{
           return true;
       }
   }
}
